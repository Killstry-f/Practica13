-- Простые запросы

SELECT Фирма_производитель, Город
FROM ОптовыеКлиенты;

SELECT Название_книги, Автор, Год_издания
FROM Книги;

SELECT Дата_заказа, Количество, Скидка
FROM Заказы;

SELECT *
FROM Книги
WHERE Год_издания = '1869';

SELECT *
FROM Книги
WHERE Автор LIKE '%Толстой%';


-- Сложные запросы

SELECT DISTINCT о.Фирма_производитель, о.Город
FROM ОптовыеКлиенты о
JOIN Заказы з ON о.Код_клиента = з.ОптовыеКлиенты_Код_клиента
WHERE з.Скидка >= 10
  AND з.Дата_заказа BETWEEN '2026-01-10' AND date('now');

SELECT з.Номер_заказа, о.Фирма_производитель, з.Дата_заказа, з.Количество, з.Скидка
FROM Заказы з
JOIN ОптовыеКлиенты о ON з.ОптовыеКлиенты_Код_клиента = о.Код_клиента
WHERE о.Фирма_производитель LIKE 'Л%'
   OR о.Фирма_производитель LIKE 'Р%';

SELECT SUM(Цена) AS Общая_сумма_книг
FROM Книги;

SELECT *
FROM Книги
WHERE Цена < 500;

SELECT о.Фирма_производитель,
       SUM(к.Цена * з.Количество) AS Потрачено_денег
FROM Заказы з
JOIN ОптовыеКлиенты о ON з.ОптовыеКлиенты_Код_клиента = о.Код_клиента
JOIN Книги к ON з.Книги_Код_книги = к.Код_книги
GROUP BY о.Фирма_производитель;
